29/* A combination of interfaces must have a unique product id, since PC will save device driver after the first plug.
30 * Same VID/PID with different interface e.g MSC (first), then CDC (later) will possibly cause system error on PC.
31 *
32 * Auto ProductID layout's Bitmap:
33 * [MSB] HID | MSC | CDC [LSB]
34 */
35#define _PID_MAP(itf, n) ( (CFG_TUD_##itf) << (n) )
36#define USB_PID (0x4000 | _PID_MAP(CDC, 0) | _PID_MAP(MSC, 1) | _PID_MAP(HID, 2) | \
37 _PID_MAP(MIDI, 3) | _PID_MAP(VENDOR, 4) )

200// Invoked when received GET OTHER SEED CONFIGURATION DESCRIPTOR request
201// Application return pointer to descriptor, whose contents must exist long enough for transfer to complete
202// Configuration descriptor in the other speed e.g if high speed then this is for full speed and vice versa
203uint8_t const* tud_descriptor_other_speed_configuration_cb(uint8_t index)
204{
205 (void) index; // for multiple configurations
206
207 // if link speed is high return fullspeed config, and vice versa
208 // Note: the descriptor type is OHER_SPEED_CONFIG instead of CONFIG
209 memcpy(desc_other_speed_config,
210 (tud_speed_get() == TUSB_SPEED_HIGH) ? desc_fs_configuration : desc_hs_configuration,
211 CONFIG_TOTAL_LEN);
212
213 desc_other_speed_config[1] = TUSB_DESC_OTHER_SPEED_CONFIG;
214
215 return desc_other_speed_config;
216}

240// array of pointer to string descriptors
241char const* string_desc_arr [] =
242{
243 (const char[]) { 0x09, 0x04 }, // 0: is supported language is English (0x0409)
244 "TinyUSB", // 1: Manufacturer
245 "TinyUSB Device", // 2: Product
246 "123456789012", // 3: Serials, should use chip ID
247 "TinyUSB CDC", // 4: CDC Interface
248 "TinyUSB MSC", // 5: MSC Interface
249};

251static uint16_t _desc_str[32];
252
253// Invoked when received GET STRING DESCRIPTOR request
254// Application return pointer to descriptor, whose contents must exist long enough for transfer to complete
255uint16_t const* tud_descriptor_string_cb(uint8_t index, uint16_t langid)
256{
257 (void) langid;
258
259 uint8_t chr_count;
260
261 if ( index == 0)
262 {
263 memcpy(&_desc_str[1], string_desc_arr[0], 2);
264 chr_count = 1;
265 }else
266 {
267 // Note: the 0xEE index string is a Microsoft OS 1.0 Descriptors.
268 // https://docs.microsoft.com/en-us/windows-hardware/drivers/usbcon/microsoft-defined-usb-descriptors
269
270 if ( !(index < sizeof(string_desc_arr)/sizeof(string_desc_arr[0])) ) return NULL;
271
272 const char* str = string_desc_arr[index];
273
274 // Cap at max char
275 chr_count = (uint8_t) strlen(str);
276 if ( chr_count > 31 ) chr_count = 31;
277
278 // Convert ASCII string into UTF-16
279 for(uint8_t i=0; i<chr_count; i++)
280 {
281 _desc_str[1+i] = str[i];
282 }
283 }
284
285 // first byte is length (including header), second byte is string type
286 _desc_str[0] = (uint16_t) ((TUSB_DESC_STRING << 8 ) | (2*chr_count + 2));
287
288 return _desc_str;
289}
