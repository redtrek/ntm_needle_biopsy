105void tud_msc_capacity_cb(uint8_t lun, uint32_t* block_count_p, uint16_t* block_size_p) {
106 TRACE_PRINTF("%s(lun=%d)\n", __func__, lun);
107 if (!tud_msc_test_unit_ready_cb(lun)) {
108 *block_count_p = 0;
109 } else {
110 DRESULT dr = disk_ioctl(lun, GET_SECTOR_COUNT, block_count_p);
111 if (RES_OK != dr) *block_count_p = 0;
112 }
113 *block_size_p = 512;
114}

128bool tud_msc_start_stop_cb(uint8_t lun, uint8_t power_condition, bool start, bool load_eject) {
129 TRACE_PRINTF("%s(lun=%d)\n", __func__, lun);
130 (void)power_condition;
131 if (load_eject) {
132 if (start) {
133 // load disk storage
134 ejected = false;
135 } else {
136 // unload disk storage
137 DRESULT dr = disk_ioctl(lun, CTRL_SYNC, 0);
138 if (RES_OK != dr) return false;
139 ejected = true;
140 }
141 }
142 return true;
143}
157int32_t tud_msc_read10_cb(uint8_t lun, uint32_t lba, uint32_t offset, void* buffer,
158 uint32_t bufsize) {
159 TRACE_PRINTF("%s(lun=%d)\n", __func__, lun);
160 (void)offset;
161 assert(!offset);
162 assert(!(bufsize % 512));
163
164 if (ejected) return -1;
165 if (!tud_msc_test_unit_ready_cb(lun)) return -1;
166
167 // Read data from the disk.
168 DRESULT dr = disk_read(lun, (BYTE*)buffer, lba, bufsize / 512);
169 if (RES_OK != dr) return -1;
170
171 return (int32_t)bufsize;
172}

228int32_t tud_msc_scsi_cb(uint8_t lun, uint8_t const scsi_cmd[16], void* buffer,
229 uint16_t bufsize) {
230 TRACE_PRINTF("%s(lun=%d)\n", __func__, lun);
231 void const* response = NULL;
232 int32_t resplen = 0;
233
234 // most scsi handled is input
235 bool in_xfer = true;
236
237 switch (scsi_cmd[0]) {
238 default:
239 // Set Sense = Invalid Command Operation
240 tud_msc_set_sense(lun, SCSI_SENSE_ILLEGAL_REQUEST, 0x20, 0x00);
241 // negative means error -> tinyusb could stall and/or response with failed status
242 resplen = -1;
243 break;
244 }
245
246 // return resplen must not larger than bufsize
247 if (resplen > bufsize) resplen = bufsize;
248
249 if (response && (resplen > 0)) {
250 if (in_xfer) {
251 memcpy(buffer, response, (size_t)resplen);
252 } else {
253 ; // SCSI output
254 }
255 }
256 return (int32_t)resplen;
257}
